# 12) 
Ripetere gli esercizi 8 e 9 usando il costrutto switch (selezione multipla)

    switch (op) 
    {
      case '+': printf("%d\n", a+b);
      break;
      case '-': printf("%d\n", a-b);
      break;
      case '*': printf("%d\n", a*b);
      break;
      case '/': printf("%d\n", a/b);
      break;
      default: printf("operatore inesistente\n");
    }
